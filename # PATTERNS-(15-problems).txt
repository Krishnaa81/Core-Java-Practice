            9 10 11 12
            16 15 14 13

#code:- 

	int n=sc.nextInt();

  	int num=1;

  	for(int i=1;i<=n;i++) {

  		if(i%2 !=0) {

  			for(int j=1;j<=n;j++) {

  				System.out.print(num+" ");
  				num++;
  			}
  		}else {
  			int temp = num+3;

  			for(int j=n;j>=1;j--) {
  				System.out.print(temp+" ");
  				temp--;
  				num++;
  			}
  		}
  		System.out.println();
  	}
